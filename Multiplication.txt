$1 answer
$2 Mcand
$3 Mplier
$4 temp
$5 = 32 endloop
$6 selectbit
$7 = 1 constant

		lw		0		2		Mcand		#$2 = Mcand
		lw		0		3		Mplier		#$3 = Mplier
		lw		0		5		eloop		#$5 = endloop
		lw		0		6		selbit		#$6 = selectbit
		lw		0		7		one		#$7 = constant value 1
		sw		7		4		loopc		#initailize countloop
		lw		0		4	    	bof         	#blockoverflow		
		sw		7		4		mof		#constant overflow value 
selMp		nand		3		6		4               #selectMplier to temp
		nand		4		4		4		#true selectMplier
		beq		0		4		shMc		#if selectMplier == 0 skip add
		add		2		1		1		#add Mcand to answer($1)
shMc		add		2		2		2		#shift Mcand
		lw		7		4		mof		#load memory overflow to $4
		nand		2		4		4		#fillterdata
		nand		4		4		2		#true fillterdata
		add		6		6		6		#shift selectbit
		lw		7		4		loopc		#load memory loopcount
		add		7		4		4		#loopcount++
		sw		7		4		loopc		#store loopcount to memory
		beq		5		4		done		#if loopcount == 31 goto done
		beq		0		0		selMp		#goto selMp
done		halt								#end program
